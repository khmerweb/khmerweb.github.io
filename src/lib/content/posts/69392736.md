    const date = formData.get("datetime")

    const validate = (
        typeof title === 'string' &amp;&amp; title !== '' &amp;&amp;
        typeof thumb === 'string' &amp;&amp; thumb !== '' &amp;&amp;
        typeof date === 'string' &amp;&amp; date !== ''
    )

    if(validate){
        const category = {title, thumb, date}
        if(user.role == 'Admin'){

}
